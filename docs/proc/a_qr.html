A_QR

public function A_QR(A, b, method) result(x)

QR decomposition method

This function implements the QR decomposition method for solving linear systems. It allows the user to choose the method for QR decomposition (e.g., Householder).

    FUNCTION A_QR(A, b, method) RESULT(x)

        REAL(dp), DIMENSION(:, :), INTENT(IN) :: A
        REAL(dp), DIMENSION(:), INTENT(IN) :: b
        CHARACTER(LEN = *), OPTIONAL, INTENT(IN) :: method
        REAL(dp), DIMENSION(SIZE(A, 1)) :: x
        REAL(dp), DIMENSION(SIZE(A, 1) ,SIZE(A, 2)) :: Q, R

        CALL QR_decomposition(A, method, Q, R)

        x = backward(R, MATMUL(TRANSPOSE(Q), b))

    END FUNCTION A_QR
