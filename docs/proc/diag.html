    FUNCTION Diag(A) RESULT(D)
        REAL(dp), DIMENSION(:, :), INTENT(IN) :: A
        REAL(dp), DIMENSION(size(A, 1)) :: D
        INTEGER :: i, N

        N = size(A, 1)

        FORALL (i=1:N) D(i) = A(i, i)

    END FUNCTION Diag
